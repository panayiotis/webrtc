    Initialize a Peer collection

    • models, is a Peer array
    • options is an object
      • server, a User.connection object. It is passed to the Peer objects that are created by the collection
      • ids, array of strings. Peers with these ids will be created on initialization. This is used by tests
      initialize : (models, options) ->
    options = options or {}
    if options.server
      if PeerJS.prototype.isPrototypeOf(options.server)
        @server = options.server
      else
        throw new Error "PeerCollection: server must be a PeerJS object"
    
    if options.ids
      @update options.ids

      @add( new App.Peer(server:@server, id:peer_id) )
